Code For Arduino

int IR_Recv = 2; //IR Receiver Pin 2
IRrecv irrecv(IR_Recv);
decode_results results;

void setup(){
pinMode(3, OUTPUT);
Serial.begin(9600); //starts serial communication
irrecv.enableIRIn(); // Starts the receiver
pinMode(4, OUTPUT);
Serial.begin(9600); //starts serial communication
irrecv.enableIRIn(); // Starts the receiver
pinMode(5, OUTPUT);
Serial.begin(9600); //starts serial communication
irrecv.enableIRIn(); // Starts the receiver
}

void loop(){
if (irrecv.decode(&results)){
long int decCode = results.value;
Serial.println(decCode);
if (results.value == 16582903) //Select button
{
digitalWrite(3, HIGH);
Serial.println(" Green LED turned ON");
}
if (results.value == 16580863) //Power button
{
digitalWrite(3, LOW);
Serial.println(" Green LED turned OFF");
}

if (results.value == 16615543) //Select button
{
digitalWrite(4, HIGH);
Serial.println(" Blue LED turned ON");
}
if (results.value == 16580863) //Power button
{
digitalWrite(4, LOW);
Serial.println(" Blue LED turned OFF");
}

if (results.value == 16599223) //Select button
{
digitalWrite(5, HIGH);
Serial.println(" Red LED turned ON");
}
if (results.value == 16580863) //Power button
{
digitalWrite(5, LOW);
Serial.println(" Red LED turned OFF");
}
irrecv.resume();
}
}
